    <button onclick="calculatePercentage()">Calcular Porcentaje</button>

<script>

    function calculatePercentage() {

        // Obtiene el valor ingresado por el usuario

        const ivInput = document.getElementById('ivValue');

        const ivValue = ivInput.value;

        const resultDiv = document.getElementById('result');


        // Validamos que el campo no esté vacío

        if (ivValue === '') {

            resultDiv.textContent = 'Por favor, ingresa un valor.';

            return;

        }


        // Convertimos el valor a un número flotante

        const iv = parseFloat(ivValue);

        const maxIV = 500;


        // Validamos que el valor esté dentro del rango esperado (0-500)

        if (iv < 0 || iv > maxIV) {

            resultDiv.textContent = 'El valor debe estar entre 0 y 500.';

            return;

        }


        // Aplicamos la fórmula para calcular el porcentaje

        const percentage = (iv / maxIV) * 100;

        const formattedPercentage = percentage.toFixed(2).replace('.', ','); // Formato con coma para el resultado


        // Mostramos el resultado en el div

        resultDiv.textContent = `El Pokémon que ingresaste es de ${formattedPercentage}%`;

    }

</script>
